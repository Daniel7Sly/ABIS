--AÇAO:
nomeaçao: parametro1|parametro2... ;
					^separador de parametros

Variaveis: 
Existem 3 tipos de Variaveis
	num 	-> com um valor numerico(pode ter ,).
	bool 	-> valor 'true' ou 'false'
	text 	-> contem texto.
Variaveis são indetificados por '$' seguido pelo no
me. Ex: $resultado

--SET
--Define uma variavel com o tipo da variavel o nome e o valor 
Set: type|nome|value;


--PRINT
Print: Value(text)\$;

--Read retorna text
Read: $;

--Define valore a variavel
Eql: id|id\value;

--Faz Operaçoes com as variaveis\valores dados
Opr: id|id\value|operador*|id\value; (c = b + a)
*operadores: + - * /

--Define flag
Flag: flag_name;

--Vai até á flag especificada
Goto: flag_name;

--Define a variavel com o resultado da comparaçao ()
CMP: $(bool)|id\value|comparator|id\value;
*comparators: < > >= <= == !=

--É um IF se true -> goto flg_nm
If: id(bool)|flag_name;
