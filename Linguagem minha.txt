Variaveis: 
Existem 3 tipos de Variaveis
	num 	-> com um valor numerico(pode ter ,).
	bool 	-> valor 'true' ou 'false'
	text 	-> contem texto.
Variaveis são indetificados por '$' seguido pelo nome. Ex: $resultado
Em arrays é usado o '#' depois do nome para indicar o index que queremos o valor. Ex: $listaNomes#23

--SET
--Define uma variavel com o tipo da variavel o nome e o valor 
Set: type|nome|value;

--SETARR
--Define um array com o tipo de dados o nome e valores/tamanho 
SetArr: type|nome|{1,2,3,4};
SetArr: type|nome|10;
SetArr: type|nome|{1, $b, $c#1};

--PRINT
--Escreve o valor na consola, '_' são substituidos por espaços
Print: Value(text)\$;

--Read retorna text
Read: $;

--Atribui o valor dado á variavel
Eql: $|$\value;

--JoinText
--junta os valores dados e atribui o resultado á variavel dada
Jtxt: $(text)|$\value|$\value;

--GetArrayLegth
--Obtem o tamanho do array dado
GLength: $(num)|$#;

--Faz Operaçoes com as variaveis\valores dados
Opr: $(num)|$\value|operador*|$\value; (c = b + a)
*operadores: + - * /

--Define flag
Flag: flag_name;

--Vai até á flag especificada
Goto: flag_name;

--Define a variavel com o resultado da comparaçao ()
CMP: $(bool)|id\value|comparator|id\value;
*comparators: < > >= <= == !=

--É um IF se true -> goto flg_nm
If: $(bool)|flag_name;
